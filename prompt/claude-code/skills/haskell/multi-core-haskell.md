データフロー並列

            並列性は自動的に抽出される

       h x = do {
           fx <- spawn (return (f x));
           gx <- spawn (return (g x));
           a <- get fx;  -- fx を待つ
           b <- get gx;  -- gx を待つ
           return (a+b)  -- 結果を返す

CAS (Compare and Swap)
 同期を取る必要がないなら CAS が使える
 CAS は CPU の不可分な命令

            ある「メモリの値」と「指定された値」を比較し、
            同じであれば「指定された別の値」で「メモリの値」を
            置き換える

39

遅延評価と CAS

 新しい値の作成を遅延させる

            「後でこれをやる」という命令書を O(1) で作成
            命令書の作成は高速なので、CAS が高い確率で成功する
            CAS が成功した後に、命令書を実行

デッドロックしない送金システム

       void
       transfer(Account from,
                Account to,
                Int amount) {
         if (from < to) then {
           lock(from); lock(to);

         deposit(to, amount);
         unlock(from); unlock(to);
       }

STM
 ソフトウェア・トランザクショナル・メモリー
 複数のロックを一つにみせる技術
            ロックを取る順番は気にしなくてよい
            部品プログラミングが可能

 STM の性質
            デッドロックフリー
            ライブロックフリーではない

 比較) ハードウェア・トランザクショナル・メモリー

            Intel の Haswell が標準提供

STM を使った送金システム

 STM の TVar はロックの代替品

       withdraw :: Account -> Int -> STM ()
       withdraw acc amount = do {
         bal <- readTVar acc;
         writeTVar acc (bal - amount)
       }

 順番を気にせず atomocally で包む

       transfer :: Account -> Account -> Int
                -> IO ()
       transfer from to amount =
         atomically $ do {
           withdraw from amount;
           deposit to amount
         }
